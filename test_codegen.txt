struct Token {
  name: string
  pos: int
}

func swap(a: var &int, b: var &int) {
  let t = *a
  *a = *b
  *b = t
}

func f(n: int) -> int {
  if n > 1 {
    return n * f(n - 1)
  } else {
    return 1
  }
}

func main() -> int {
  let a = 1
  let b = 4
  swap(var &a, var &b)
  #printf("%d\n", a)

  // structs
  let tok = Token {.name = "comma", .pos = 256}
  #printf("%s, %d\n", tok.name, tok.pos)
  return 0
}
